<script>
(() => {
  // --- State ---
  let measure = "RR";              // RR | HR
  let pattern = "regular";         // regular | irregular
  let secondsMode = 15;            // default 15s
  let running = false;
  let taps = 0;
  let timerId = null;
  let endAt = null;

  // Age group
  let ageGroup = "adult";          // adult | paeds

  // --- Elements ---
  const measureSeg = document.getElementById("measureSeg");
  const patternSeg = document.getElementById("patternSeg");
  const timeSeg = document.getElementById("timeSeg");
  const irregularBanner = document.getElementById("irregularBanner");
  const statusBanner = document.getElementById("statusBanner");

  const ageGroupSeg = document.getElementById("ageGroupSeg");
  const paedsControls = document.getElementById("paedsControls");
  const ageBanner = document.getElementById("ageBanner");
  const ageBandBanner = document.getElementById("ageBandBanner");
  const ageValue = document.getElementById("ageValue");
  const ageUnit = document.getElementById("ageUnit");

  const tapBtn = document.getElementById("tapBtn");
  const startBtn = document.getElementById("startBtn");
  const undoBtn = document.getElementById("undoBtn");
  const resetBtn = document.getElementById("resetBtn");
  const tapCount = document.getElementById("tapCount");
  const timeRemaining = document.getElementById("timeRemaining");
  const resultText = document.getElementById("resultText");
  const resultMeta = document.getElementById("resultMeta");

  // --- Helpers ---
  function setActiveButtons(container, predicateFn) {
    [...container.querySelectorAll("button")].forEach(btn => {
      btn.classList.toggle("active", predicateFn(btn));
    });
  }

  function ageToDays(value, unit) {
    const v = Math.max(0, Number.isFinite(value) ? value : 0);
    if (unit === "days") return v;
    if (unit === "months") return v * 30.44;
    return v * 365.25;
  }

  // RADAR paeds bands:
  // Neonate 0–7d; <3m; 3m–1y; 1–4y; 5–11y; 12–16y
  function radarPaedsBandLabel(days) {
    if (days <= 7) return "Neonate (0–7 days)";
    if (days < (3 * 30.44)) return "<3 months";
    if (days < 365.25) return "3 months – 1 year";
    if (days < (5 * 365.25)) return "1–4 years";
    if (days < (12 * 365.25)) return "5–11 years";
    return "12–16 years";
  }

  function updatePaedsBandUI() {
    if (ageGroup !== "paeds") return;
    const v = parseFloat(ageValue.value);
    const u = ageUnit.value;
    const days = ageToDays(v, u);
    ageBandBanner.textContent = radarPaedsBandLabel(days);
    ageBanner.innerHTML = `Age group: <b>Paeds</b> (${ageBandBanner.textContent}).`;
  }

  function applyAgeGroupUI() {
    paedsControls.classList.toggle("hide", ageGroup !== "paeds");
    if (ageGroup === "paeds") updatePaedsBandUI();
    else ageBanner.innerHTML = `Age group: <b>Adult</b>.`;
    updateStatusReady();
  }

  function lockTimerTo60IfIrregular() {
    const isIrreg = pattern === "irregular";
    irregularBanner.classList.toggle("hide", !isIrreg);

    // Disable duration selection when irregular
    [...timeSeg.querySelectorAll("button")].forEach(btn => btn.disabled = isIrreg);

    if (isIrreg) {
      secondsMode = 60;
      setActiveButtons(timeSeg, b => b.dataset.seconds === "60");
    }
    updateStatusReady();
  }

  function updateStatusReady() {
    if (running) return;
    const s = secondsMode;
    const what = (measure === "RR") ? "breath" : "beat";
    const pat = (pattern === "irregular") ? "Irregular" : "Regular";
    statusBanner.className = "banner";
    statusBanner.innerHTML = `Ready: <b>${measure}</b> (${pat}), window <b>${s}s</b>. Press <b>Start</b>, then tap each ${what}.`;
  }

  function updateUI() {
    tapCount.textContent = String(taps);
    undoBtn.disabled = !running || taps === 0;
    tapBtn.disabled = !running;
    startBtn.textContent = running ? "Stop" : "Start";
    resetBtn.disabled = !running && (taps === 0 && resultText.textContent.startsWith("--"));
  }

  function computeRate(tapsCount, secondsCounted) {
    const rate = tapsCount * (60 / secondsCounted);
    return Math.round(rate);
  }

  function finishCount(fullWindowSeconds, note) {
    running = false;
    if (timerId) clearInterval(timerId);
    timerId = null;

    const rate = computeRate(taps, fullWindowSeconds);
    const pat = (pattern === "irregular") ? "Irregular" : "Regular";

    resultText.textContent = `${rate} bpm`;

    let ageMeta = (ageGroup === "paeds") ? `Paeds (${ageBandBanner.textContent})` : "Adult";
    resultMeta.textContent = `${measure} • ${pat} • ${ageMeta} • ${fullWindowSeconds}s • taps=${taps}${note ? " • " + note : ""}`;

    statusBanner.className = note ? "banner warn" : "banner good";
    statusBanner.innerHTML = note ? note : `Complete. ${measure} calculated from ${taps} taps over ${fullWindowSeconds}s.`;

    timeRemaining.textContent = "0";
    updateUI();
  }

  function tick() {
    const msLeft = endAt - Date.now();
    const secLeft = Math.max(0, Math.ceil(msLeft / 1000));
    timeRemaining.textContent = String(secLeft);
    if (msLeft <= 0) finishCount(secondsMode, "");
  }

  // --- Handlers ---
  measureSeg.addEventListener("click", (e) => {
    const btn = e.target.closest("button[data-measure]");
    if (!btn || running) return;
    measure = btn.dataset.measure;
    setActiveButtons(measureSeg, b => b.dataset.measure === measure);
    tapBtn.textContent = (measure === "RR") ? "Tap for each breath" : "Tap for each beat";
    updateStatusReady();
    updateUI();
  });

  patternSeg.addEventListener("click", (e) => {
    const btn = e.target.closest("button[data-pattern]");
    if (!btn || running) return;
    pattern = btn.dataset.pattern;
    setActiveButtons(patternSeg, b => b.dataset.pattern === pattern);
    lockTimerTo60IfIrregular();
    updateUI();
  });

  timeSeg.addEventListener("click", (e) => {
    const btn = e.target.closest("button[data-seconds]");
    if (!btn || running) return;
    if (pattern === "irregular") return; // locked
    secondsMode = parseInt(btn.dataset.seconds, 10);
    setActiveButtons(timeSeg, b => b.dataset.seconds === btn.dataset.seconds);
    updateStatusReady();
  });

  ageGroupSeg.addEventListener("click", (e) => {
    const btn = e.target.closest("button[data-agegroup]");
    if (!btn || running) return;
    ageGroup = btn.dataset.agegroup;
    setActiveButtons(ageGroupSeg, b => b.dataset.agegroup === ageGroup);
    applyAgeGroupUI();
  });

  ageValue.addEventListener("input", () => { if (!running && ageGroup === "paeds") updatePaedsBandUI(); });
  ageUnit.addEventListener("change", () => { if (!running && ageGroup === "paeds") updatePaedsBandUI(); });

  tapBtn.addEventListener("click", () => {
    if (!running) return;
    taps += 1;
    tapCount.textContent = String(taps);
    undoBtn.disabled = taps === 0;
    if (navigator.vibrate) navigator.vibrate(10);
  });

  undoBtn.addEventListener("click", () => {
    if (!running || taps === 0) return;
    taps -= 1;
    tapCount.textContent = String(taps);
    undoBtn.disabled = taps === 0;
  });

  resetBtn.addEventListener("click", () => {
    if (running) {
      taps = 0;
      tapCount.textContent = "0";
      undoBtn.disabled = true;
      statusBanner.className = "banner";
      statusBanner.innerHTML = "Count reset. Continue tapping or press Stop to finish early.";
      return;
    }
    taps = 0;
    timeRemaining.textContent = "--";
    resultText.textContent = "-- bpm";
    resultMeta.textContent = "";
    updateStatusReady();
    updateUI();
  });

  startBtn.addEventListener("click", () => {
    if (!running) {
      // START
      taps = 0;
      tapCount.textContent = "0";
      resultText.textContent = "-- bpm";
      resultMeta.textContent = "";

      running = true;
      endAt = Date.now() + (secondsMode * 1000);
      timeRemaining.textContent = String(secondsMode);

      statusBanner.className = "banner";
      statusBanner.innerHTML = `Counting for <b>${secondsMode}s</b>. Tap each ${(measure === "RR") ? "breath" : "beat"}.`;

      timerId = setInterval(tick, 100);
      updateUI();
      return;
    }

    // STOP EARLY (allowed)
    const sTotal = secondsMode;
    const elapsed = Math.max(1, Math.round((sTotal * 1000 - (endAt - Date.now())) / 1000));
    finishCount(elapsed, `Stopped early at <b>${elapsed}s</b>. Result uses elapsed time.`);
  });

  // --- Init ---
  tapBtn.textContent = "Tap for each breath";
  applyAgeGroupUI();
  lockTimerTo60IfIrregular();
  updateStatusReady();
  updateUI();
})();
</script>
